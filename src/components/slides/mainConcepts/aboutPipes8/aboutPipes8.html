Main Concepts Pipes - Custom Ones

  • And as for services and directives, you can create your own custom Pipes
  • Use the @Pipe annotation
  • Implement the PipeTransform interface
  • Below is an example of "Hello World" Pipe
import {Component, Pipe, PipeTransform} from '@angular/core'; import {LowerCasePipe} from '@angular/common'; import {bootstrap} from 'angular2/platform/browser'; import {SubCmp} from './sub-cmp'; @Pipe({ name:'helloWorld' }) class HelloWorldPipe implements PipeTransform{ transform(value: string, args:any[] = null){ return 'Hello ' + value; } } @Component({ selector: 'my-app', template: ` <p> {{'John' | helloWorld }}</p> `, pipes: [HelloWorldPipe] }) export class AppComponent {} bootstrap(AppComponent), []);
