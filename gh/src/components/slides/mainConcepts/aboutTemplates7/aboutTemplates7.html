Main Concepts About Templates - Property Binding - About Expansions

  • Expansions are done with the {{expression}}
  • They allow to evaluate Angular2 expressions
  • They are a shortcut for property bindings on the textContent property
import {Component} from '@angular/core'; import {bootstrap} from 'angular2/platform/browser'; import {SubCmp} from './sub-cmp'; @Component({ selector: 'my-app', template: '<sub-cmp name="john"></sub-cmp>', directives: [SubCmp] }) export class AppComponent { name: string= 'john'; } bootstrap(AppComponent), []); import {Component, Input} from '@angular/core'; @Component({ selector: 'sub-cmp', template: ` <h1> Name v1: {{name}}</h1> <h1 [textContent]="'Name v2: ' + name" > </h1> ` }) export class SubCmp { @Input() name: string; }
