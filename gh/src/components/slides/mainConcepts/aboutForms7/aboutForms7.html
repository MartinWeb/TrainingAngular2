Main Concepts Forms - Custom Validators

  • Firstly, we need a function which implements this interface interface Validator<T extends Control> { (c:T): {[error: string]:any}; }
  • Then we can use it with our ControlGroup programmatically
  • You can also declare it as a directive if needed (more here)
import {Component} from '@angular/core'; import {bootstrap} from 'angular2/platform/browser'; import {MyFormCmp} from 'app/components/form-cmp'; @Component({ selector: 'my-app', template: '<my-form-cmp></my-form-cmp>', directives: [MyFormCmp] }) export class AppComponent {} bootstrap(AppComponent), []); import {Component} from '@angular/core'; import {FormBuilder, ControlGroup, Control, Validators} from '@angular/common'; enum TodoCategory{PRIVATE, PERSO}; // THE VALIDATOR FUNCTION let users= ['John']; function userNameExists(c:Control){ for (var i in users){ if ( users[i] === c.value ){ return null; } } return {userNameExists: true }; } @Component({ selector:'my-form-cmp', templateUrl:'app/components/form-cmp.html' }) export class MyFormCmp{ isSubmit: false; userValue: string; customControlGroup: ControlGroup; constructor(fb: FormBuilder){ this.customControlGroup= fb.group({ 'user': new Control('', Validators.compose([Validators.required, userNameExists])) }); } onSubmit(){ this.isSubmit= true; } } <h1>Add an element to your todo list</h1> <form (submit)="onSubmit()" [ngFormModel]="customControlGroup" > <p> <label for="user"> User </label> <input [ngFormControl]="customControlGroup.controls?.user" [(ngModel)]="userValue" id="user" type="text"> </p> <p> <span [hidden]="customControlGroup.controls.user?.valid">User name is not valid!</span> </p> <input type="submit" value="send" [disabled]="!customControlGroup.valid" > </form> <p *ngIf="isSubmit"> Form submited </p>
