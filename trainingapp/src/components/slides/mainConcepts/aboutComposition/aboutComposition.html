Main Concepts About Composition

  • Remember our Web Comboments composition purpose with the content tag?
  • Angular allows to do the same with the ng-content attribute
  • You can even get references on the content with @ContentChild and @ContentChildren
  • Note that you can also use @ViewChild and @ViewChildren to get access to your template children (no composition, just the child components of your current components)
import {Component, ContentChildren, AfterViewInit} from 'angular2/core'; import {bootstrap} from 'angular2/platform/browser'; @Component({ selector: 'person', template: `hello {{name}}`, inputs: ['name'], }) class Person { name: string; } @Component({ selector: 'display-persons', template: ` <ng-content></ng-content> <p> We are displaying {{persons?.length}} persons </p> ` }) class DisplayPersons implements AfterViewInit{ @ContentChildren(Person) persons: QueryList<Person>; } @Component({ selector: 'my-app', template: ` <display-persons> <ul> <li><person [name]="'john'"></person></li> <li><person [name]="'bob'"></person></li> </ul> </display-persons> `, directives: [DisplayPersons, Person] }) export class AppComponent {} bootstrap(AppComponent), []);
