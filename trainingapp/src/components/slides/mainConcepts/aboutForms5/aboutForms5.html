Main Concepts Forms - Controls and ControlGroups

  • Let's now reproduce the previous sample but with our own ControlGroup
  • FormBuilder let us create this ControlGroup dynamically
  • ngFormModel allows to bind our custom ControlGroup to the ngForm directive

This way, we can also easily manipulate data / controls

import {Component} from '@angular/core'; import {bootstrap} from 'angular2/platform/browser'; import {MyFormCmp} from 'app/components/form-cmp'; @Component({ selector: 'my-app', template: '<my-form-cmp></my-form-cmp>', directives: [MyFormCmp] }) export class AppComponent {} bootstrap(AppComponent), []); import {Component} from '@angular/core'; import {FormBuilder, ControlGroup} from '@angular/common'; enum TodoCategory{PRIVATE, PERSO}; class TodoModel{ constructor(public category: TodoCategory, public details: string){} } @Component({ selector:'my-form-cmp', templateUrl:'app/components/form-cmp.html' }) export class MyFormCmp{ categories: string[]= Object.keys(TodoCategory).filter(v => isNaN(parseInt(v, 10))); model: TodoModel= new TodoModel(); isSubmit: false; customControlGroup: ControlGroup; constructor(fb: FormBuilder){ this.customControlGroup= fb.group({ 'details':'' }) } onSubmit(){ this.isSubmit= true; } } <h1>Add an element to your todo list</h1> <form (submit)="onSubmit()" [ngFormModel]="customControlGroup" > <p> <label for="category"> Category </label> <select [(ngModel)]="model.category" id="category" required> <option *ngFor="#category of categories" [value]="category">{{category}}</option> </select> </p> <p> <label for="todoDetails">Content</label> <input required [ngFormControl]="customControlGroup.controls?.details" [(ngModel)]="model.details" id="todoDetails" type="text"> <span [hidden]="customControlGroup.controls.details?.valid">I miss data!</span> </p> <input type="submit" value="send" disabled="!customControlGroup.valid" > </form> <p *ngIf="isSubmit"> Form submited </p> <h2>Result</h2> <p *ngIf="!isSubmit"> {{model.category}} : {{model.details}} </p>
