Main Concepts About Templates - Event Binding - Use your events

  • You can easily create your own events
  • Use next() method on an EventEmitter instance
  • This allows a down to top communication when associated with outputs attribute in the @Component annotation (or the @Output annotation directly)
import {bootstrap} from 'angular2/platform/browser'; import {AppComponent} from 'app/components/app'; bootstrap(AppComponent, []); import {Component} from 'angular2/core'; import {SubCmp} from './subcmp'; @Component({ selector: 'my-app', template: ` <h1>Checking event propagation</h1> <p (click)="logEvent('click', $event)" > <sub-cmp (customevent)="logEvent('custom', $event)" ></sub-cmp> </p>`, directives: [SubCmp] }) export class AppComponent { logEvent(msg, event){ console.log(msg); console.log(event); } } import {Component, EventEmitter, Output} from 'angular2/core'; @Component({ selector:'sub-cmp', template: ` <span (click)="propagateCustomEvent($event)" > Click Here </span>` }) export class SubCmp{ @Output() customevent= new EventEmitter(); constructor(){} propagateCustomEvent(){ console.log('propagate'); this.customevent.next('some data'); } }
