Main Concepts Router - Url Parameters

  • It's also possible to add parameters to your urls
  • To do so, use the following path syntax: /path/to/route/:id
  • You can put several parameters as well
  • The target component can then use this (these) parameter(s) to achieve its task
  • Use the RouteParams service to get these parameters
import {Component, provide} from 'angular2/core'; import {bootstrap} from 'angular2/platform/browser'; //Router dependencies import {RouteConfig, ROUTER_PROVIDERS, LocationStrategy, HashLocationStrategy, ROUTER_DIRECTIVES} from 'angular2/router'; // Component Views Dependencies import {TodoListCmp} from 'app/components/todolist-cmp'; import {TodoFormCmp} from 'app/components/todoform-cmp'; import {TodoDetailsCmp} from 'app/components/tododetails-cmp'; @Component({ selector: 'my-app', template: ` <nav> <ul> <li><a [routerLink]="['TodoList']">Go to Todo List</a></li> <li><a [routerLink]="['TodoForm']">Go to Todo Form</a></li> </ul> </nav> <router-outlet></router-outlet> `, directives: [ROUTER_DIRECTIVES] }) @RouteConfig([ {path:'/todolist', name:'TodoList', component:TodoListCmp}, {path:'/todoform', name:'TodoForm', component:TodoFormCmp}, {path:'/todo/:id', name:'TodoDetails', component:TodoDetailsCmp} ]) export class AppComponent {} let ALL_BINDINGS= [ ROUTER_PROVIDERS, provide(LocationStrategy, {useClass: HashLocationStrategy}), provide('TODOS', {useValue:[{id: 1, label: 'todo 1'}, {id: 2, label: 'todo 2'}] }) ]; bootstrap(AppComponent, [ALL_BINDINGS]); import {Component, Inject} from 'angular2/core'; import {ROUTER_DIRECTIVES} from 'angular2/router'; @Component({ selector:'todolist', template:` <h1>Todo List</h1> <p>...</p> <ul> <li *ngFor="#todo of todos"> <a [routerLink]="['TodoDetails', {id: todo.id}]">{{todo.id}} {{todo.label}}</a> </li> </ul> `, directives: [ROUTER_DIRECTIVES] }) export class TodoListCmp{ public todos: {id: int, label: string}; constructor(@Inject('TODOS') public todos:{id: int, label: string}){ } } import {Component} from 'angular2/core'; @Component({ selector:'todoform', template:` <h1>Todo Form</h1> <p>...</p> ` }) export class TodoFormCmp{} import {Component, Inject, OnInit} from 'angular2/core'; import {RouteParams} from 'angular2/router'; @Component({ selector:'tododetails', template:` <h1>Todo Detail</h1> <ul> <li>Id: {{todo.id}}</li> <li>Label: {{todo.label}}</li> </ul> ` }) export class TodoDetailsCmp implements OnInit{ todo: {id: int, label: string}; } } ); } }
