Main Concepts Pipes - Exisiting Ones - I18nSelectPipe

  • Use a map to retrieve values (not necessarily related to the language)
  • Type name: I18nSelectPipe
  • Expression name: i18nSelect
  • mapping: the maps to use
import {Component} from '@angular/core'; import {bootstrap} from 'angular2/platform/browser'; import {SubCmp} from './sub-cmp'; @Component({ selector: 'my-app', template: ` <p> Message: {{ getLocalizedKey('msg') | i18nSelect:map }} </p> <button (click)="switchLocale()">switch locale</button> ` }) export class AppComponent { public availableLocales: string[]= ['fr', 'en']; private index: number= 0; public map: any = { 'msg.fr':'Bienvenue'; 'msg.en':'Welcome'; }; switchLocale(){ this.index= (++this.index) % 2; } getLocalizedKey(key:string){ return key + '.' + this.availableLocales[this.index]; } } bootstrap(AppComponent), []);
