Main Concepts Dependency Injection - More Details - Provide

And we can even do more with provide

  • configure simple values (singletons)
  • configure aliases (singletons)
  • configure factories
import {Injector, provide} from 'angular2/core'; //Define our injector and its providers let injector = Injector.resolveAndCreate([ provide('AppVersion', {useValue: '1.0.0-SNAPSHOT'}) ]); import {Injector, provide} from 'angular2/core'; //Define our injector and its providers let injector = Injector.resolveAndCreate([ provide(Person, {useClass: PersonImpl}), provide(MainUser, {useExisting: Person}) ]); import {Injector, provide} from 'angular2/core'; //Define our injector and its providers let injector = Injector.resolveAndCreate([ provide(PersonFactory, { useFactory: (loggerService) => { return new RandomPerson(loggerService); }, deps: [LoggerService] }), ]);
