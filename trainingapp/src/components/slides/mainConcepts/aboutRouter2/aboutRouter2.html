Main Concepts Router - Configuration

  • The configuration allows to define this mapping
  • It can be done either
    • By using the @RouteConfig decorator
    • By using the Router service
  • Once configuration is done, you still need
    • To define the zone where the view will be displayed
    • A way to navigate from a view to another

Below is a first example of routing defined at the main application component level

import {Component} from 'angular2/core'; import {bootstrap} from 'angular2/platform/browser'; //Add router dependencies import {RouteConfig, ROUTER_PROVIDERS} from 'angular2/router'; // Component Views Dependencies import {TodoListCmp} from 'app/components/todolist-cmp'; import {TodoFormCmp} from 'app/components/todoform-cmp'; @Component({ selector: 'my-app', template: ` <nav> <ul> <li><a [routerLink]="['TodoList']">Go to Todo List</a></li> <li><a [routerLink]="['TodoForm']">Go to Todo Form</a></li> </ul> </nav> `, directives: [] }) @RouteConfig([ {path:'/todolist', name:'TodoList', component:TodoListCmp}, {path:'/todoform', name:'TodoForm', component:TodoFormCmp} ]) export class AppComponent {} bootstrap(AppComponent), [ROUTER_PROVIDERS]); import {Component} from 'angular2/core'; @Component({ selector:'todolist', template:` <h1>Todo List</h1> <p>...</p> ` }) export class TodoListCmp{} import {Component} from 'angular2/core'; @Component({ selector:'todoform', template:` <h1>Todo Form</h1> <p>...</p> ` }) export class TodoFormCmp{}
